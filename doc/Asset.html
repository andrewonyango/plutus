def self.balance
  accounts_balance = BigDecimal.new('0') 
  accounts = self.find(:all)
  accounts.each do |asset|
    unless asset.contra
      accounts_balance += asset.balance
    else
      accounts_balance -= asset.balance
    end
  end
  accounts_balance
end
